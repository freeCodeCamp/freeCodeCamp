### --question--

#### --text--

Which of the following is the correct way to prevent a browser refresh for a `onSubmit` event?

#### --distractors--

```jsx
function handleSubmit(event) {
  event.defaultPrevent();
  console.log("Form submitted!");
}

<form onSubmit={handleSubmit}>
  <input type="text" />
  <button>Submit</button>
</form>;
```

---

```jsx
function handleSubmit(event) {
  event.Default();
  console.log("Form submitted!");
}

<form onSubmit={handleSubmit}>
  <input type="text" />
  <button>Submit</button>
</form>;
```

---

```jsx
function handleSubmit(event) {
  event.prevent();
  console.log("Form submitted!");
}

<form onSubmit={handleSubmit}>
  <input type="text" />
  <button>Submit</button>
</form>;
```

#### --answer--


```jsx
function handleSubmit(event) {
  event.preventDefault();
  console.log("Form submitted!");
}

<form onSubmit={handleSubmit}>
  <input type="text" />
  <button>Submit</button>
</form>;
```
