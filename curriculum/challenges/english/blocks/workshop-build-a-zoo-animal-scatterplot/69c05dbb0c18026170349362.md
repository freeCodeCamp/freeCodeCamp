# --description--

**Let's add legends**  
We have done our core part, now we will add some descriptive features to our scatterplot. Let's start with legends. 

```js
const legend = svg
        .selectAll(".legend")
        .data(color.domain())
        .enter()
        .append("g")
        .attr("class", "legend")
        .attr("transform", (d, i) => `translate(0,${i * 20})`);
```

A selection named legend is created, that is binded to the array *["Mammal", "Bird", "Reptile", "Amphibian"]*. A group *g* with legend class is appended to each element of that legend selection. Different index of that selection has different *translate* value along y axis, that will create the stack of legends.

```js
legend
  .append("rect")
  .attr("x", width - 18)
  .attr("width", 18)
  .attr("height", 18)
  .style("fill", (d) => color(d));
```

Then a `rectangle` element of *18×18* size is appended to each *group*, and colored by color scale. It's location is 
(width-18, 0) from (0,0) origin point.

```js
legend
  .append("text")
  .attr("x", width - 24)
  .attr("y", 9)
  .attr("dy", ".35em")
  .style("text-anchor", "end")
  .style("font-size", windowWidth > 700 ? "16px" : "12px")
  .text((d) => d);
```

Now, a text element with each text inside the array is appended to each *group*, with location [width - 24, 9], slightly left and middle(y-axis) of each rectangle. Then, little adjustment is made using "dy" attribute.  

Now, add legends of the color scale at top left the xy-plane.

```js
const margin = { top: 60, right: 60, bottom: 50, left: 60 };
const maxHeight = 500;
const maxWidth = 800;

const svg = d3
  .select("body")
  .append("svg")
  .attr("class", "graph")
  .append("g")
  .attr("transform", `translate(${margin.left},${margin.top})`);

const color = d3
  .scaleOrdinal()
  .domain(["Mammal", "Bird", "Reptile", "Amphibian"])
  .range(["#1f77b4", "#ff7f0e", "#2ca02c", "#d62728"]);

  

function drawScatterplot() {
  const windowWidth = window.innerWidth;
  let width = Math.min(maxWidth, windowWidth * 0.9) - margin.left - margin.right;
  let height = Math.min(maxHeight, window.innerHeight * 0.7) - margin.top - margin.bottom;

  d3.select("svg")
    .attr("width", width + margin.left + margin.right)
    .attr("height", height + margin.top + margin.bottom);
  
  svg.selectAll("*").remove();

  let x, y, xAxis, yAxis;

  d3.json("https://cdn.freecodecamp.org/curriculum/workshop-build-a-zoo-animal-scatterplot/data.json").then((data) => {

    if (windowWidth > 700) {

      x = d3
        .scaleLinear()
        .domain([0, d3.max(data, (d) => d.Weight) + 100])
        .range([0, width]);
      y = d3
        .scaleLinear()
        .domain([0, d3.max(data, (d) => d.Lifespan) + 10])
        .range([height, 0]);
    } else {

      x = d3
        .scaleLinear()
        .domain([0, d3.max(data, (d) => d.Lifespan) + 10])
        .range([0, width]);
      y = d3
        .scaleLinear()
        .domain([0, d3.max(data, (d) => d.Weight) + 100])
        .range([height, 0]);
    }

    xAxis = d3.axisBottom(x);
    yAxis = d3.axisLeft(y);

    svg
      .append("g")
      .attr("transform", `translate(0,${height})`)
      .call(xAxis)
      .append("text")           
      .attr("x", width / 2)    
      .attr("y", 40)           
      .style("text-anchor", "middle")
      .style("fill", "black")   
      .style("font-size", windowWidth > 700 ? "16px" : "12px")
      .text(windowWidth > 700 ? "Weight (kg)" : "Lifespan (years)");

    svg
      .append("g")
      .call(yAxis)
      .append("text")
      .attr("transform", "rotate(-90)")    
      .attr("x", -height / 2)             
      .attr("y", -margin.left + 15)        
      .style("text-anchor", "middle")      
      .style("fill", "black")              
      .style("font-size", windowWidth > 700 ? "16px" : "12px")
      .text(windowWidth > 700 ? "Lifespan (years)" : "Weight (kg)");

      --fcc-editable-region--

      --fcc-editable-region--
  });
}

drawScatterplot();
window.addEventListener("resize", drawScatterplot);
```
