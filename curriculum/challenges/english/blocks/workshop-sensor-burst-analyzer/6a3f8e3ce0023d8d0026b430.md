  }

}

function getWindow(burst, start, windowSize) {
  const sample = [...burst];

  if (sample.length == 0) {
    return "burst is empty";
  }

  if (sample.length <= windowSize) {
    return "lecture is smaller than the window size";
  }
  
  return sample.slice(start, start + windowSize);
}
