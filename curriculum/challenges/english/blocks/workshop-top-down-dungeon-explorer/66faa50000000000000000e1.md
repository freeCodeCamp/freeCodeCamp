class UIScene extends Phaser.Scene {
  constructor() {
    super({ key: "UIScene", active: false });
  }

  create() {
    const main = this.scene.get("MainScene");
    this.box = this.add
      .rectangle(160, 210, 304, 44, 0x000000, 0.7)
      .setVisible(false);
    this.box.setStrokeStyle(1, 0xffffff, 0.4);
    this.text = this.add.text(16, 192, "", {
      fontSize: "10px",
      color: "#ffffff",
      wordWrap: { width: 288 }
    });
    this.text.setVisible(false);
    main.events.on("ui:dialogue", (line) => {
      this.fullLine = line;
      this.shown = 0;
      this.typing = true;
      this.text.setText("").setVisible(true);
      this.box.setVisible(true);
    });
    this.fullLine = "";
    this.shown = 0;
    this.typing = false;
    this.notice = this.add.text(160, 24, "", {
      fontSize: "10px",
      color: "#ffe066",
      backgroundColor: "rgba(0,0,0,0.7)",
      padding: { x: 4, y: 2 }
    });
    this.notice.setOrigin(0.5, 0.5).setVisible(false);
    main.events.on("ui:notice", (msg) => {
      this.notice.setText(msg);
      this.notice.setVisible(true);
      this.time.delayedCall(2000, () => this.notice.setVisible(false));
    });
  }

  update(time, delta) {
    if (this.typing) {
      this.shown += delta * 0.06;
      const slice = this.fullLine.slice(0, Math.floor(this.shown));
      this.text.setText(slice);
      if (slice.length >= this.fullLine.length) this.typing = false;
    }
  }
}
