```py
def array_diff(arr1, arr2):
    unique1 = []
    for v in arr1:
        if v not in unique1:
            unique1.append(v)
    unique2 = []
    for v in arr2:
        if v not in unique2:
            unique2.append(v)

    only1 = [v for v in unique1 if v not in unique2]
    only2 = [v for v in unique2 if v not in unique1]

    return sorted(only1 + only2)
```
