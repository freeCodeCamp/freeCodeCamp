# --description--

CSV (Comma-Separated Values) is a widely used format for spreadsheets and data import. Building a CSV string from structured data means constructing output by processing each element in a collection, and we will once more a `for` loop for that purpose.

Create a function named `exportToCSV` that takes `catalog` as a parameter. Declare a variable `header` set to `"Title,Author,Year,Location"` and an empty array `rows`.

Use a `for` loop to iterate over `catalog`. For each entry, wrap the string fields in double quotes to handle commas within the data (for example, `"King, Stephen"` contains a comma that would break the CSV format without quotes). Push each row as a comma-separated string into `rows`.

After the loop, declare `let csv = header`. Use another `for` loop to append each row to `csv` with a newline separator: `csv = csv + "\n" + rows[i]`. Return `csv`.

Below the function, log `exportToCSV(catalog)` to see the result.

function parseCard(rawString) {
  const parts = rawString.split("|");
  const trimmedParts = [];
  for (let i = 0; i < parts.length; i++) {
    trimmedParts.push(parts[i].trim());
  }
  const title = trimmedParts[0];
  const author = trimmedParts[1];
  const year = trimmedParts[2];
  const location = trimmedParts[3];
  return {
    title: title || "Unknown",
    author: author || "Unknown",
    year: year ? parseInt(year) : "Unknown",
    location: location || "Unknown"
  };
}

function parseCatalog(rawCards) {
  const catalog = [];
  for (let i = 0; i < rawCards.length; i++) {
    catalog.push(parseCard(rawCards[i]));
  }
  return catalog;
}

function groupByDecade(catalog) {
  const grouped = {};
  for (let i = 0; i < catalog.length; i++) {
    const book = catalog[i];
    if (book.year === "Unknown") {
      if (!grouped["Unknown"]) {
        grouped["Unknown"] = [];
      }
      grouped["Unknown"].push(book);
      continue;
    }
    const decade = Math.floor(book.year / 10) * 10;
    const decadeKey = `${decade}s`;
    if (!grouped[decadeKey]) {
      grouped[decadeKey] = [];
    }
    grouped[decadeKey].push(book);
  }
  return grouped;
}
